    <script>
        // Eenvoudige functionaliteit om emoji's en speciale tekens te verwijderen
        document.addEventListener('DOMContentLoaded', function() {
            const inputTextarea = document.querySelector('.text-panel:first-child textarea');
            const outputTextarea = document.querySelector('.text-panel:last-child textarea');

            inputTextarea.addEventListener('input', function() {
                // Verwijder emoji's en speciale tekens
                // Dit is een eenvoudige versie die alleen basistekens behoudt
                const cleanText = inputTextarea.value.replace(/[^\x00-\x7F]/g, '');
                outputTextarea.value = cleanText;
            });
        });
